def load_images_from_folder(folder):
    images = []
    labels = []
    label_map = {}
    label = 0
    
    for filename in os.listdir(folder):
        path = os.path.join(folder, filename)
        img = cv2.imread(path)
        
        if img is not None:
            img = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)  # Convert to RGB
            img = cv2.resize(img, (64, 64))  # Resize image to 64x64
            images.append(img)
            labels.append(label)
            
            label_map[label] = filename.split('.')[0]  # Map label to gesture name
            label += 1
            
    images = np.array(images)
    labels = np.array(labels)
    
    return images, labels, label_map

def resize_and_save_images(input_dir, output_dir, size=(128, 128)):
    if not os.path.exists(output_dir):
        os.makedirs(output_dir)
    labels = {}
    for idx, folder in enumerate(os.listdir(input_dir)):
        folder_path = os.path.join(input_dir, folder)
        save_folder = os.path.join(output_dir, folder)
        if not os.path.exists(save_folder):
            os.makedirs(save_folder)
        labels[folder] = idx  # Assign numeric label to each folder
        for img_name in os.listdir(folder_path):
            img_path = os.path.join(folder_path, img_name)
            try:
                img = cv2.imread(img_path)
                img_resized = cv2.resize(img, size)
                cv2.imwrite(os.path.join(save_folder, img_name), img_resized)
            except Exception as e:
                print(f"Error processing {img_path}: {e}")
    return labels

def create_model(input_shape, num_classes):
    model = models.Sequential([
        layers.Conv2D(32, (3, 3), activation='relu', input_shape=input_shape),
        layers.MaxPooling2D((2, 2)),
        layers.Conv2D(64, (3, 3), activation='relu'),
        layers.MaxPooling2D((2, 2)),
        layers.Conv2D(128, (3, 3), activation='relu'),
        layers.MaxPooling2D((2, 2)),
        layers.Flatten(),
        layers.Dense(128, activation='relu'),
        layers.Dense(num_classes, activation='softmax')
    ])
    model.compile(optimizer='adam', loss='sparse_categorical_crossentropy', metrics=['accuracy'])
    return model
